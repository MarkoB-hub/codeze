#version 330 core

out vec4 color;

in vec2 vTexCoords;
flat in int vTexIndex;
in vec4 vColor;

uniform sampler2D uTextures[3];

void main() {
	color = vColor;
	if (vTexIndex == 0) {
		color *= texture(uTextures[0], vTexCoords);
	}
	else if (vTexIndex == 1) {
		color *= texture(uTextures[1], vTexCoords);
	}
	else if (vTexIndex == 2) {
		color *= texture(uTextures[2], vTexCoords);
	}
}
