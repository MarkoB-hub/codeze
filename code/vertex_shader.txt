#version 330 core

layout (location = 0) in vec4 aPosData;
layout (location = 1) in vec4 aColor;
layout (location = 2) in float aTexIndex;

out vec2 vTexCoords;
out float vTexIndex;
out vec4 vColor;

uniform mat4 uProjection;

void main() {
	gl_Position = uProjection * vec4(aPosData.xy, 0.0, 1.0);
	vTexCoords = aPosData.zw;
	vTexIndex = aTexIndex;
	vColor = aColor;
}
